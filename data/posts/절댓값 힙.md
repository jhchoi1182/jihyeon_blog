최소값 출력 후 제거

```
import sys
import heapq

# 빠른 입력 함수 사용
input = sys.stdin.readline

# 최소 힙 초기화
heap = []

# 연산의 개수 입력
n = int(input())

for _ in range(n):
    x = int(input())  # 연산 정보 입력

    if x == 0:  # 삭제 연산
        if len(heap) == 0:  # 힙이 비어있는 경우
            print(0)
        else:  # 힙에 원소가 있는 경우
            absolute, original = heapq.heappop(heap)  # 가장 작은 절대값 원소 삭제
            print(original)  # 원래 수 출력
    else:  # 삽입 연산
        heapq.heappush(heap, (abs(x), x))  # 절대값과 원래 값을 함께 힙에 삽입
```
