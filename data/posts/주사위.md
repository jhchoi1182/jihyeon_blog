https://www.acmicpc.net/problem/1233

3개의 주사위를 던졌을 때 가장 많이 나오는 경우의 수 찾기

```
import sys

# 빠른 입력 함수 사용
input = sys.stdin.readline

# 세 주사위의 면 개수 각각 입력받기
s1, s2, s3 = map(int, input().split())
counter = dict()

# 3중 반복문을 이용해 모든 경우의 수 계산
for i in range(1, s1 + 1):
    for j in range(1, s2 + 1):
        for k in range(1, s3 + 1):
            summary = i + j + k # 눈금의 합
            # 각 합의 등장 횟수 카운트
            if summary not in counter:
                counter[summary] = 1
            else:
                counter[summary] += 1

# 가장 많이 등장한 합 찾기
max_count = -1  # 존재할 수 없는 수로 세팅
answer = int(1e9) # 10억
for (key, value) in counter.items():
    # 등장 횟수(빈도수)가 더 많은 경우를 찾은 경우
    if max_count < value:
        max_count = value
        answer = key
    elif max_count == value:
        # 빈도수가 동일한 것이 여러 개라면, 가장 작은 것 출력
        answer = min(answer, key)

print(answer)

```
